        # UploadArea
        Renders a drag-and-drop file upload surface with click-to-select fallback. It also supports optional rendering of a file list beneath the drop zone so users can see files currently uploading or already selected.

        <RenderSource language={"tsx"}>
{`
import { UploadArea } from '@apple-pie/slice/uikit/UploadArea';

export function UploadAreaExample() {
  return (
    <UploadArea
      title="Upload files"
      message="Drag files here or click to browse"
      acceptedTypes={['pdf', 'png', 'jpg']}
      multiple
      busy={false}
      showProgress
      files={[
        { file: 'proposal.pdf', uploading: true, progress: 0.45 },
        { file: 'brief.clipboard' },
      ]}
      canRemove
      onUpload={(files) => {
        console.log('Selected files:', files);
      }}
      id="upload-area"
      data-testid="upload-area"
      style={{ maxWidth: 520 }}
    />
  );
}
`}
