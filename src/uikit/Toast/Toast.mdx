        # Toast
        Renders transient status messaging with optional auto-dismiss and close action.

        ## Singleton usage pattern
        Add a single `Toast` instance for the app (or app shell), not one per component. Use shared state management to set `message` so that one toast handles notifications across the UI.

        ## Base element
        `motion.div`

        ## Interface
        `ReservedToastEvents = 'onMouseEnter' | 'onMouseLeave' | 'onFocus' | 'onBlur'`
        `Omit<HTMLMotionProps<'div'>, keyof ToastBaseProps | ReservedToastEvents> & ToastBaseProps`

        <RenderSource language={"tsx"}>
{`
import { useState } from 'react';
import { Toast, Button } from '@apple-pie/slice';

export function ToastStatusExample() {
  const [message, setMessage] = useState<string | null>(null);

  return (
    <>
      <Button
        label="Save profile"
        onClick={() => setMessage('Profile saved successfully')}
      />

      <Toast
        message={message}
        type="success"
        close={true}
        position="bottom"
        didHide={() => setMessage(null)}
        id="profile-save-toast"
        data-testid="profile-save-toast"
        aria-live="polite"
      />
    </>
  );
}
`}
        </RenderSource>

import { useState } from 'react';
import { Toast, IconButton } from '@apple-pie/slice';

export function DelayedWarningToast() {
  const [message, setMessage] = useState<string | null>(null);

  return (
    <>
      <IconButton
        icon="alert-triangle"
        tooltip="Trigger warning toast"
        onClick={() => setMessage('Connection is unstable')}
      />

      <Toast
        message={message}
        type="warning"
        showDelay={300}
        duration={4000}
        close={false}
        position="top"
        offset={20}
        didHide={() => setMessage(null)}
        className="network-toast"
        style={{ maxWidth: 420 }}
      />
    </>
  );
}
`}
