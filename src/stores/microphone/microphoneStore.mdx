        # MicrophoneStore
        Global microphone state helpers built with Zustand. Mirrors a live `useMicrophone()` instance into a module-level singleton store and exposes atomic selector hooks plus imperative actions.

        ## Example
        </Unstyled>

        <RenderSource language={"tsx"}>
        ```
        import { useMicrophone } from '@apple-pie/slice';
        import {
            useMicMuted,
            useMicActive,
            useMicTrack,
            useMicrophoneStoreActions,
            useSyncMicrophoneStore,
        } from '@apple-pie/slice/stores/microphone';

        function GlobalMicrophoneProvider() {
            const microphone = useMicrophone(true, undefined, false);
            useSyncMicrophoneStore(microphone);
            return null;
        }

        function MicrophoneStatus() {
            const isActive = useMicActive();
            const muted = useMicMuted();
            const micTrack = useMicTrack();
            const actions = useMicrophoneStoreActions();

            return (
                <div>
                    <div>Active: {String(isActive)}</div>
                    <div>Muted: {String(muted)}</div>
                    <div>Track: {micTrack.current?.readyState ?? 'none'}</div>
                    <button type="button" onClick={() => void actions.requestMicrophone()}>
                        Request Microphone
                    </button>
                </div>
            );
        }
        ```
