        # UploadsStore
        Global upload queue state built with Zustand and a Web Worker for optimal performance allowing app interactions while uploads are in progress.
        The store tracks upload state, while the worker performs the upload scheduling, progress tracking, etc. on a separate thread.

        ## Important!
        `uploads store` requires initialization with the uploads web worker instance. It does not create the worker automatically and defers to the consuming app's tooling to do so.
        
        ## Web worker setup
        Slice ships as a standalone uploads worker file. Depending on your project setup, you have a few basic setup options:
        - Use the provided `createUploadsWorker()` with no argument for local/package-internal setups.
        - Pass an explicit worker URL into `createUploadsWorker(workerUrl)` when your app wants to control worker resolution.
            
        For app-level integrations, the most portable pattern is `new URL('./uploads.ts', import.meta.url)`. That works well across Vite, Vue, Webpack, Rspack, Angular, and most browser-first bundlers. Frameworks with SSR such as Next.js should create the worker only on the client.
        
        <RenderSource language={"tsx"}>
{`
// note: for SSR frameworks such as Next.js, create the worker only on the client.
export function createUploadsWorker(workerUrl?: string | URL) {
  return new Worker(
    workerUrl ?? new URL('../../workers/uploads/uploads.ts', import.meta.url),
    { type: 'module' },
  );
}
`}
        </RenderSource>    

       Here's an example of how you might implement the uploads store with a singleton worker instance:

        <RenderSource language={"tsx"}>
```
import React, { useEffect, useRef } from 'react';
import {
  createUploadsWorker,
  useUploadsActions,
  useUploadsWorkerStatus,
} from '@apple-pie/slice/stores/uploads';

// setup uploads
const config = {
    uploadURL: '/api/uploads',
    maxQueueSize: 10,
    maxConcurrentUploads: 2,
};

// create a worker
function getStoryWorker(): UploadsWorkerInstance | null {
    const uploadsWorkerUrl = new URL('../../workers/uploads/uploads.ts', import.meta.url);
    return createUploadsWorker(uploadsWorkerUrl);
}

// initialize the store and use atomic hook throughout your app
export function App() {
    const actions = useUploadsActions();
    const workerStatus = useUploadsWorkerStatus();
    const workerRef = useRef<Worker | null>(null);

    useEffect(() => {
        if (workerRef.current) return;
        workerRef.current = getStoryWorker();
        workerRef.current && actions.initialize(storyOptions, workerRef.current);
    }, [actions]);

    return <div>{workerStatus}</div>;
}
```
        </RenderSource>

        ## Pre-signed URLs
        Many applications upload directly to object storage using a pre-signed URL instead of sending every file to one shared endpoint. The uploads worker supports that by allowing each queued file to include an optional `uploadURL`.

import { useUploadsActions } from '@apple-pie/slice/stores/uploads';

function PresignedUploadButton({

  presignedUrl,

  presignedUrl: string;

  const actions = useUploadsActions();

        ## React usage with `createUploadsWorker`
        This pattern works well for local package development and internal tooling such as Storybook. Create the worker once, initialize the store once, and then push files into the queue.

        <RenderSource language={"tsx"}>
{`
import { useEffect } from 'react';
import {
  createUploadsWorker,
  useUploads,
  useUploadsActions,
  useUploadsError,
  useUploadsWorkerStatus,
} from '@apple-pie/slice/stores/uploads';

// Create the uploads worker once for the lifetime of your app.
const worker = createUploadsWorker();

function UploadManager() {
  const uploads = useUploads();
  const workerStatus = useUploadsWorkerStatus();
  const error = useUploadsError();
  const actions = useUploadsActions();

  useEffect(() => {
    // Initialize the uploads store once so the singleton store and worker stay in sync.
    actions.initialize(
      {
        uploadURL: '/api/uploads',
        maxQueueSize: 20,
        maxConcurrentUploads: 3,
        accepted: ['image/png', 'image/jpeg', 'pdf'],
      },
      worker,
    );
  }, [actions]);

  return (
    <div>
      <div>status: {workerStatus}</div>
      <div>error: {error ?? 'none'}</div>
      <input
        type="file"
        multiple
        onChange={(event) => {
          if (!event.target.files) return;
          actions.push(Array.from(event.target.files));
        }}
      />
      <ul>
        {uploads.map((upload) => (
          <li key={upload.id}>
            {upload.file.name} - {upload.status} - {upload.progress ?? 0}%
          </li>
        ))}
      </ul>
    </div>
  );
}
`}
        </RenderSource>

        ## Consuming app usage with a built worker asset
        In consuming apps, the most reliable approach is to pass an explicit worker URL into `createUploadsWorker(workerUrl)`.
        <br />
        Prefer `new URL('./uploads.ts', import.meta.url)` as the default portable pattern. If your bundler supports `?url` imports, that can also work, but it is bundler-specific and may need extra TypeScript module declarations.

        <RenderSource language={"tsx"}>
{`
import { createUploadsWorker, uploadsActions } from '@apple-pie/slice/stores/uploads';

const worker = createUploadsWorker(new URL('./uploads.ts', import.meta.url));

uploadsActions.initialize(
  {
    uploadURL: '/api/uploads',
    maxQueueSize: 10,
    maxConcurrentUploads: 2,
  },
  worker,
);
`}
        </RenderSource>

        Vite-style `?url` example:

        <RenderSource language={"tsx"}>
{`
import { createUploadsWorker, uploadsActions } from '@apple-pie/slice/stores/uploads';
// @ts-expect-error Vite resolves ?url imports, but TypeScript needs an explicit declaration for them.
import uploadsWorkerUrl from '@apple-pie/slice/workers/uploads/uploads?url';

// Create the uploads worker once for the lifetime of your app.
const worker = createUploadsWorker(uploadsWorkerUrl);

// Initialize the uploads store once with the shared worker instance.
uploadsActions.initialize(
  {
    uploadURL: '/api/uploads',
    maxQueueSize: 10,
    maxConcurrentUploads: 2,
  },
  worker,
);
`}
        </RenderSource>

        ## Imperative usage outside React
        Use imperative helpers for app bootstrap, service layers, and non-component workflows.

        <RenderSource language={"tsx"}>
{`
import {
  createUploadsWorker,
  uploadsActions,
  getUploads,
  getUploadsWorkerStatus,
} from '@apple-pie/slice/stores/uploads';

// Create the uploads worker once for the lifetime of your app.
const worker = createUploadsWorker();

export function initializeUploadSystem() {
  // Initialize the uploads store once during app bootstrap.
  uploadsActions.initialize({ uploadURL: '/api/uploads' }, worker);
}

export function pushIncomingFiles(files: File[]) {
  uploadsActions.push(files);
}

export function clearUploadActivity() {
  uploadsActions.reset();
}

export function getUploadSnapshot() {
  return {
    status: getUploadsWorkerStatus(),
    uploads: getUploads(),
  };
}
`}
