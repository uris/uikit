        # Tip
        Renders a positioned tooltip surface and animates visibility based on resolved coordinates.

        ## Singleton usage pattern
        Add a single `Tip` instance for the app (or app shell), not one per component. Use shared state management to set `tip` and `coords` from any element emitting tooltip events, so one global `Tip` handles show/hide across the full UI.
            
        ## Base element
        `HTMLDivElement`

        ## Interface
        `ToolTipProps`

        <RenderSource>
```
import { useRef, useState } from 'react';
import { IconButton, Tip, useToolTip } from '@apple-pie/slice';
import type { ToolTip } from '@apple-pie/slice';

function TipExample() {
    const [tip, setTip] = useState<ToolTip | null>(null);
    const tipRef = useRef<HTMLDivElement>(null);
    const coords = useToolTip(tip, tipRef);

    return (
        <>
            <IconButton
                icon="home"
                tooltip="Click Me"
                onToolTip={(nextTip) => setTip(nextTip)}
            />
            <Tip
                id="tip-example"
                data-testid="tip-example"
                className="custom-wrapper"
                style={{ zIndex: 20 }}
                ref={tipRef}
                tip={tip}
                coords={coords}
                size="m"
            />
        </>
    );
}
```
