        # RadioButtonList
        Renders a typed list of `RadioButton` options and manages single or multi-selection state.

        ## Base element
        `HTMLDivElement`

        ## Interface
        `RadioButtonListProps<T = string>`

        ## Related types
        `RadioButtonOption<T = string>`

        ## Typing
        `RadioButtonList` is generic. The `options` array and the `onChange` callback use the same `T`.

        `onChange` receives:

        `(options: RadioButtonOption<T>[] | null, indexes: number[] | null) => void`

        ## Notes
        - `selectedIndexes` and `selectedOptions` both drive controlled selection.
        - `selectedOptions` matches against each option's `fieldName`.
        - In single-select mode, the root uses `role="radiogroup"`.
        - In multi-select mode, the root uses `role="group"` and children render as checkbox-style controls.

```
import { RadioButtonList } from '@apple-pie/slice';

export function RadioButtonListExample() {
    return (
        <RadioButtonList
            label="Notification Channels"
            options={[
                { label: 'Email', fieldName: 'email', value: 'email' },
                { label: 'SMS', fieldName: 'sms', value: 'sms' },
            ]}
            selectedIndexes={[0]}
            onChange={(options, indexes) => console.log(options, indexes)}
            id="radiobuttonlist-example"
            data-testid="radiobuttonlist-example"
            className="custom-wrapper"
            style={{ marginTop: 8, maxWidth: 480 }}
        />
    );
}
```

```
import { RadioButtonList, type RadioButtonOption } from '@apple-pie/slice';

type DeliveryValue = {
    id: number;
    method: 'email' | 'sms';
};

const options: RadioButtonOption<DeliveryValue>[] = [
    {
        fieldName: 'delivery-email',
        label: 'Email',
        value: { id: 1, method: 'email' },
    },
    {
        fieldName: 'delivery-sms',
        label: 'SMS',
        value: { id: 2, method: 'sms' },
    },
];

export function TypedRadioButtonListExample() {
    return (
        <RadioButtonList<DeliveryValue>
            label="Delivery Method"
            options={options}
            onChange={(selectedOptions, selectedIndexes) => {
                console.log(selectedOptions?.[0]?.value?.method, selectedIndexes);
            }}
        />
    );
}
```
