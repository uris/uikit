        # Video
        Renders an HTML video element with optional built-in controls, custom controls, progress scrubbing, volume control, and an imperative ref API for playback actions.

        ## Base element
        `HTMLDivElement`

        ## Interface
        `VideoProps`

        ## Ref Interface
        `VideoElement`

```
import { useRef } from 'react';
import { Video, type VideoElement } from '@apple-pie/slice';

export function VideoExample() {
    const videoRef = useRef<VideoElement>(null);

    return (
        <Video
            ref={videoRef}
            src="https://example.com/video.mp4"
            controls="simple"
            width="100%"
            height="auto"
            muted={false}
            customControls={{
                play: true,
                progress: true,
                volume: true,
                fullscreen: true,
            }}
            onPlayStateChange={(state) => console.log('state', state)}
            onFullScreenChange={(fullScreen) => console.log('fullscreen', fullScreen)}
            onVolumeChange={(volume) => console.log('volume', volume)}
            style={{ maxWidth: 720 }}
        />
    );
}
```

import { useRef } from 'react';
import { Button, Video, type VideoElement } from '@apple-pie/slice';

export function VideoRefExample() {
    const videoRef = useRef<VideoElement>(null);

    return (
        <>
            <Button onClick={() => videoRef.current?.play?.()}>Play</Button>
            <Button onClick={() => videoRef.current?.stop?.()}>Pause</Button>
            <Button onClick={() => videoRef.current?.restart?.()}>Restart</Button>
            <Button onClick={() => videoRef.current?.fullscreen?.(true)}>Fullscreen</Button>

            <Video
                ref={videoRef}
                src="https://example.com/video.mp4"
                controls="none"
                muted
            />
        </>
    );
}
