                { icon: 'more', action: 'more', tip: 'More actions' },
            ]}
            selected={0}
            onChange={(button) => console.log(button)}
            id="buttonbar-example"
            data-testid="buttonbar-example"
            className="custom-wrapper"
            style={{ marginTop: 8, maxWidth: 480 }}
        />
    );
}
