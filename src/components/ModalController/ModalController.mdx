        # ModalController
        Renders the active global modal from `ModalStore` and places an overlay beneath it.

        Mount one `ModalController` near the app root when using the modal store. The controller owns overlay rendering, enter and exit animation, and optional dragging.

        ## Base element
        `HTMLDivElement`

        ## Interface
        `ModalControllerProps`

        ## Behavior
        `ModalController` reads the current modal from the modal store and injects `onResolve`, `onReject`, `onClose`, and `onDragPointerDown` into the rendered modal component.

        ## Code Examples
        </Unstyled>

        <RenderSource language={"tsx"}>
```
import { useRef } from 'react';
import { Button, FlexDiv, Modal, ModalController } from '@apple-pie/slice';
import { useModalActions } from '@apple-pie/slice/stores/modal';

export function ModalControllerExample() {
    const actions = useModalActions();
    const boundsRef = useRef<HTMLDivElement>(null);

    return (
        <FlexDiv ref={boundsRef} height={320} justify="center" align="center">
            <Button
                onClick={() =>
                    actions.show({
                        component: Modal,
                        props: {
                            title: 'Confirmation',
                            children: 'Continue?',
                        },
                    })
                }
            >
                Open Modal
            </Button>
            <ModalController dragConstraintsRef={boundsRef} />
        </FlexDiv>
    );
}
```

        ## Drag Constraints
        Use `dragConstraintsRef` to keep the active modal within another element's bounds.
        </Unstyled>

        <RenderSource language={"tsx"}>
```
import { useRef } from 'react';
import { FlexDiv, ModalController } from '@apple-pie/slice';

export function ConstrainedControllerExample() {
    const ref = useRef<HTMLDivElement>(null);

    return (
        <FlexDiv ref={ref} height={320} justify="center" align="center">
            <ModalController dragConstraintsRef={ref} />
        </FlexDiv>
    );
}
```
