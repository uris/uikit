        # VideoController
        Renders the active global video from `VideoStore` and places an overlay beneath it.

        Mount one `VideoController` near the app root when using the video store. The controller owns overlay rendering, enter and exit animation, optional dragging, and a small readiness gate that prevents a size flash before the first frame loads.

        ## Base element
        `HTMLDivElement`

        ## Interface
        `VideoControllerProps`

        ## Behavior
        `VideoController` reads the current video descriptor from the video store and renders its `component` with the provided `props`.

        It also injects two controller-owned props into the rendered video component:
        - `borderRadius`, unless the active video already provides one
        - `onLoadedFrameData`, which is used internally to reveal the controller once the first frame is ready, and then forwards to any caller-provided `onLoadedFrameData`

        ## Code Examples
        </Unstyled>

        <RenderSource language={"tsx"}>
```
import { Button, FlexDiv, Video, VideoController } from '@apple-pie/slice';
import { useVideoActions } from '@apple-pie/slice/stores/video';

export function VideoControllerExample() {
    const actions = useVideoActions();

    return (
        <FlexDiv justify="center" align="center" height={320}>
            <Button
                onClick={() =>
                    actions.show({
                        component: Video,
                        props: {
                            src: 'https://example.com/video.mp4',
                            controls: 'simple',
                            muted: false,
                            width: '100%',
                            height: 'auto',
                        },
                    })
                }
            >
                Open Video
            </Button>
            <VideoController />
        </FlexDiv>
    );
}
```

        ## Drag Constraints
        Use `dragConstraintsRef` when the active video should stay within another element's bounds.
        </Unstyled>

        <RenderSource language={"tsx"}>
```
import { useRef } from 'react';
import { FlexDiv, VideoController } from '@apple-pie/slice';

export function ConstrainedVideoControllerExample() {
    const ref = useRef<HTMLDivElement>(null);

    return (
        <FlexDiv ref={ref} height={360} justify="center" align="center">
            <VideoController dragConstraintsRef={ref} />
        </FlexDiv>
    );
}
```
