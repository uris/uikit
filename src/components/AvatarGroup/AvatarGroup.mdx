        ## Code Examples

        <RenderSource language={"tsx"}>
```
import { AvatarGroup } from '@apple-pie/slice';

const avatars = [
    {
        first: 'John',
        last: 'Appleseed',
        image: '/images/profile-male-02.jpg',
        email: 'john@email.com',
    },
    {
        first: 'Jane',
        last: 'Appleseed',
        email: 'jane@email.com',
        bgColor: 'var(--core-surface-tertiary)',
    },
];

export function TeamAvatars() {
    return (
        <AvatarGroup
            avatars={avatars}
            size={38}
            overlap={8}
            borderSize={3}
            outerBorderSize={3}
        />
    );
}
```
