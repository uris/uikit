        # AvatarGroup
        Renders a stacked list of avatars for participants, assignees, or collaborators.

        ## Base element
        `HTMLDivElement`

        ## Interface
        `React.HTMLAttributes<HTMLDivElement> & AvatarGroupProps`

            <tr><td><code>border</code></td><td><code>number</code></td><td><code>3</code></td><td>Border width applied to each child avatar.</td></tr>

        ## Code Examples
        </Unstyled>

        <RenderSource language={"tsx"}>
{`
import { AvatarGroup } from '@apple-pie/slice';

const avatars = [
  {
    first: 'John',
    last: 'Appleseed',
    image: '/images/profile-male-02.jpg',
    email: 'john@email.com',
  },
  {
    first: 'Jane',
    last: 'Appleseed',
    image: '',
    email: 'jane@email.com',
  },
];

export function TeamAvatars() {
  return <AvatarGroup avatars={avatars} size={38} overlap={8} border={3} />;
}
`}
