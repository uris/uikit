        # DropDown
        `DropDown` renders a styled select control with typed options, controlled selection by index or value, and a typed `onChange` callback.

        ## Base element
        `HTMLDivElement`

        ## Typing
        `DropDown` is generic and defaults to string values:

        <RenderSource language={"ts"}>
        ```
        type DropDownOption<T = string> = {
            label?: string;
            value?: T;
            alt?: string;
        };
        
        type DropDownProps<T = string> = {
            label?: string;
            labelColor?: string;
            options?: DropDownOption<T>[];
            selectedValue?: T;
            valueKey?: string;
            borderStyle?: 'bottom' | 'box' | 'none';
            onChange?: (index: number, option: DropDownOption<T>) => void;
            // ...other visual props
        };
        ```
        </RenderSource>

        Use string values when possible. If option values are objects, pass `valueKey` so `selectedValue` can match options by a simple field such as `id`.

        ## Selection behavior
        `DropDown` resolves selection in this order:
        - If `selectedValue` is provided, it tries to match by direct value equality first.
        - If `valueKey` is provided and the values are objects, it compares `option.value[valueKey]` to `selectedValue[valueKey]`.
        - As a fallback, it compares the string form of `selectedValue` against `label` and `alt`.
        - If `selectedValue` is not provided, `selectedIndex` controls the selection.
        - If neither is provided, the component falls back to internal index state initialized from `selectedIndex ?? 0`.

        <RenderSource language={"tsx"}>
        ```
        import { DropDown } from '@apple-pie/slice';
        
        export function BasicDropDownExample() {
            return (
                <DropDown
                    name="Jurisdiction"
                    label="Jurisdiction"
                    borderStyle="box"
                    options={[
                        { label: 'California', value: 'ca' },
                        { label: 'New York', value: 'ny' },
                    ]}
                    selectedValue="ny"
                    onChange={(index, option) => console.log(index, option.value)}
                    id="dropdown-example"
                    data-testid="dropdown-example"
                    className="custom-wrapper"
                    style={{ marginTop: 8, maxWidth: 480 }}
                />
            );
        }
        ```
        </RenderSource>

        <RenderSource language={"tsx"}>
        ```
        import { DropDown, type DropDownOption } from '@apple-pie/slice';
        
        type UserOptionValue = {
            id: number;
            name: string;
        };
        
        const options: DropDownOption<UserOptionValue>[] = [
            { label: 'Alice', value: { id: 1, name: 'Alice' } },
            { label: 'Bob', value: { id: 2, name: 'Bob' } },
        ];
        
        export function ObjectValueDropDownExample() {
            return (
                <DropDown<UserOptionValue>
                    name="User"
                    label="User"
                    options={options}
                    valueKey="id"
                    selectedValue={{ id: 2, name: 'Bob' }}
                    onChange={(index, option) => {
                        console.log(index, option.value?.id);
                    }}
                />
            );
        }
        ```
