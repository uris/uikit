        # useToolTip
        Computes tooltip coordinates from an anchor ref and the rendered tooltip rectangle.
        The resolved position stays within the viewport with a 10px inset, and flips above the anchor when there is not enough space below.

        ## Signature
        `useToolTip(toolTip, tipElement)`

        ## Behavior
        - Centers the tooltip beneath the anchor by default.
        - Clamps horizontal placement so the tooltip remains within the viewport with a 10px margin.
        - Repositions the tooltip above the anchor when showing below would overflow the viewport bottom edge.

        ## Code Examples
        </Unstyled>

        <RenderSource language={"tsx"}>
        ```
        import { useRef, useState } from 'react';
        import { IconButton, Tip, useToolTip } from '@apple-pie/slice';
        import type { ToolTip } from '@apple-pie/slice';
        
        function TipExample() {
            const [tip, setTip] = useState<ToolTip | null>(null);
            const tipRef = useRef<HTMLDivElement>(null);
            const coords = useToolTip(tip, tipRef);
        
            return (
                <>
                    <IconButton
                        icon="home"
                        tooltip="Click Me"
                        onToolTip={(nextTip) => setTip(nextTip)}
                    />
                    <Tip ref={tipRef} tip={tip} coords={coords} size="m" />
                </>
            );
        }
        ```
