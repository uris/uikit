        # useLastUpdated
        Takes any given timestamp, compares it to the current time, and provides a human-readable text signaling the elapsed time. By default, the elapsed time text is updated every minute (configurable).

        ## Signature
        `useLastUpdated(timestamp, prefix?, interval?)`

        <RenderSource language={"tsx"}>

import { useLastUpdated } from '@apple-pie/slice';

function UpdatedAt({ timestamp }: { timestamp?: string }) {
  const { lastUpdated } = useLastUpdated(timestamp, 'Updated ', 1);

  return (
    <span style={{ color: 'var(--core-text-secondary)' }}>
      {lastUpdated || 'No updates yet'}
    </span>
  );
}
