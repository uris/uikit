        # useWindow
        Tracks viewport size, form factor, touch capability, pixel ratio, and Electron runtime state.

        ## Signature
        `useWindow(breakpoints?: BreakPoints, top?: boolean)`

        ## Parameters
        <table style={{width: '100%'}}>

                <col style={{width: '97%'}} />
            </colgroup>
            <thead>
            <tr>
                <th>Parameter</th>
                <th>Type</th>
                <th>Default</th>
                <th>Description</th>
            </tr>
            </thead>
            <tbody>
            <tr><td><code>breakpoints?</code></td><td><code>BreakPoints</code></td><td><code>{`{ mobile: 576, tablet: 768, desktop: 992, desktopL: 1200, desktopXL: 1400 }`}</code></td><td>Optional width thresholds used to resolve <code>formFactor</code>. If omitted, the hook uses its built-in defaults (Bootstrap 5 breakpoints).</td></tr>
            <tr><td><code>top?</code></td><td><code>boolean</code></td><td><code>true</code></td><td>Use the top most window in the window hierarchy. If used in an iFrame that's on a different domain, it will fall back to that iFrame window.</td></tr>

import { useWindow } from '@apple-pie/slice';

function ViewportBadge() {
  const win = useWindow();

  return (
    <div style={{ padding: 12, border: '1px solid var(--core-outline-primary)' }}>
      {win.formFactor} • {win.viewportWidth} x {win.viewportHeight}
    </div>
  );
}
`}

        <RenderSource language={"tsx"}>
{`
import { FormFactor, useWindow, type BreakPoints } from '@apple-pie/slice';

const customBreakpoints: BreakPoints = {
  [FormFactor.Mobile]: 600,
  [FormFactor.Tablet]: 1024,
  [FormFactor.Desktop]: 1280,
  [FormFactor.DesktopL]: 1536,
  [FormFactor.DesktopXL]: 1920,
};

function ViewportBadgeWithCustomBreakpoints() {
  const win = useWindow(customBreakpoints);

  return (
    <div style={{ padding: 12, border: '1px solid var(--core-outline-primary)' }}>
      {win.formFactor} • {win.viewportWidth} x {win.viewportHeight}
    </div>
  );
}
`}
