        # useKeyboardShortcuts
        Adds a global keydown listener and triggers handlers for configured `Cmd/Ctrl + key` combinations.

        ## Signature
        `useKeyboardShortcuts(shortcuts, shortCutHandler, isAppleDevice)`

import { useWindow, useKeyboardShortcuts } from '@apple-pie/slice';

const SHORTCUTS = [
    { key: 'b', name: 'bold' },
    { key: 's', name: 'save' },
]

function Shortcuts() {
    const { isAppleDevice } = useWindow();
    useKeyboardShortcuts(shortcuts, handleShortcut, isAppleDevice);

    function handleShortcut(shortcut: KeyboardShortcut) {
        console.log('shortcut', shortcut.name);
    }
    
    return <div>Try Cmd/Ctrl + B or Cmd/Ctrl + S</div>;
}
