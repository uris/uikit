        # useKeyboardShortcuts
        Adds a global keydown listener and triggers handlers for configured shortcuts.
        Shortcut matching is skipped when focus is inside editable elements (`input`, `textarea`, `select`, or content-editable).

        ## Signature
        `useKeyboardShortcuts(shortcuts, shortCutHandler, isAppleDevice)`

        <RenderSource language={"tsx"}>
{`
import { useWindow, useKeyboardShortcuts } from '@apple-pie/slice';
import type { KeyboardShortcut, KeyboardShortcuts } from '@apple-pie/slice';

const shortcuts: KeyboardShortcuts = [
	{ key: 'b', name: 'Bold', metaPressed: true },
	{ key: 's', name: 'Save', metaPressed: true },
	{ key: 'k', name: 'Quick Open', metaPressed: true, shiftPressed: true },
	{ key: 'd', name: 'Duplicate', altPressed: true },
	{ key: 'a', name: 'A Pressed' },
];

function Shortcuts() {
	const { isAppleDevice } = useWindow();
	useKeyboardShortcuts(shortcuts, handleShortcut, isAppleDevice);

	function handleShortcut(shortcut: KeyboardShortcut) {
		console.log('shortcut', shortcut.name);
	}

	return (
		<div>
			Try Cmd/Ctrl + B, Cmd/Ctrl + S, Cmd/Ctrl + Shift + K, Alt/Option + D, or A
		</div>
	);
}
`}
