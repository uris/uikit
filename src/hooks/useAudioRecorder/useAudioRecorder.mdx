        # useAudioRecorder
        Records audio from a `MediaStream` or a single `MediaStreamTrack` using `MediaRecorder` and returns the finished audio blob plus simple recording state.

        ## Behavior
        The hook does not request microphone access itself. It records whatever stream or track is passed in, which makes it pair naturally with `useMicrophone`.

        ## Signature
        `useAudioRecorder(audioStream)`

        ## Basic example
        </Unstyled>

        <RenderSource language={"tsx"}>
        ```
        import { useEffect, useState } from 'react';
        import { useMicrophone, useAudioRecorder } from '@apple-pie/slice';

        function AudioCapture() {
            const { micStream } = useMicrophone(false);
            const { isRecording, startRecording, stopRecording } = useAudioRecorder(
                micStream.current,
            );
            const [audioUrl, setAudioUrl] = useState<string | null>(null);

            useEffect(() => {
                return () => {
                    if (audioUrl) URL.revokeObjectURL(audioUrl);
                };
            }, [audioUrl]);

            return (
                <div>
                    <div>Recording: {String(isRecording)}</div>
                    <button type="button" onClick={() => startRecording()}>
                        Start
                    </button>
                    <button
                        type="button"
                        onClick={async () => {
                            const blob = await stopRecording();
                            if (!blob) return;
                            if (audioUrl) URL.revokeObjectURL(audioUrl);
                            setAudioUrl(URL.createObjectURL(blob));
                        }}
                    >
                        Stop
                    </button>
                    {audioUrl && <audio controls src={audioUrl} />}
                </div>
            );
        }
        ```
