        # useObserveResize
        Tracks an `HTMLElement` element's `width` and `height` using `ResizeObserver`, optionally ignoring width or height resize events.

        ## Signature
        `useObserveResize(element, options?)`

        ## Code Examples
        </Unstyled>

        <RenderSource language={"tsx"}>
{`
import { useRef } from 'react';
import { useObserveResize } from '@apple-pie/slice';

function BoxInfo() {
  const ref = useRef<HTMLDivElement>(null);
  const size = useObserveResize(ref, { ignore: 'height' });

  return (
    <div>
      <div ref={ref} style={{ width: 320, height: 120 }} />
      <p>Width only updates: {size.width}</p>
    </div>
  );
}
`}
